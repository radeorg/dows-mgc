# 替换secrets.txt中的密文
## 方式一
java -jar bfg.jar --replace-text secrets.txt --no-blob-protection .git





## 方式二
pip install git-filter-repo
git-filter-repo --replace-text secrets.txt --force

git reflog expire --expire=now --all
git gc --prune=now --aggressive

git remote add origin git@github.com:radeorg/dows-mgc.git
git push --force --all


git push --force origin dev-1.0.251026
